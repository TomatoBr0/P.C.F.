javascript:(function(){
  let iframe = document.querySelector('iframe#interactiveIframe');
  let w = iframe ? iframe.contentWindow : window;

  let progress = w.progressLevelList || null;
  if (!progress) {
    alert('No game state (progressLevelList) found!');
    return;
  }

  // Fill all groups as mastered
  if (Array.isArray(progress)) {
    progress.fill(2);
  }

  // Try to grab any stored answer list
  let answers =
    w.SRList ||
    w.SRListA ||
    w.SRListM ||
    w.SRListW ||
    w.QAList ||
    null;

  let level = w.userlevel || '(unknown)';
  let group = w.pickedGroup ?? '(unknown)';
  let question = w.selectedQuestion ?? '(unknown)';

  alert(
    'ALL GROUPS MASTERED!\n' +
    'Level: ' + level + '\n' +
    'Group: ' + group + '\n' +
    'Question: ' + question + '\n\n' +
    'Progress: ' + JSON.stringify(progress) + '\n\n' +
    'Answers (SRList/QAList):\n' + (answers ? JSON.stringify(answers) : 'None found')
  );
})();
